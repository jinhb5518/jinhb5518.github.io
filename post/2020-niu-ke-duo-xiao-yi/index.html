const int N = 1e6+200;

char ss[N];int s[N];
int n, sa[N], rk[N], oldrk[N << 1], id[N], px[N], cnt[N];
// px[i] = rk[id[i]]（用于排序的数组所以叫 px）

bool cmp(int x, int y, int w) {
  return oldrk[x] == oldrk[y] && oldrk[x + w] == oldrk[y + w];
}
void get_sa()
{
	  

  int i, m = n, p, w;
  for(int i=0;i<=m;i++)cnt[i]=0;
  for (i = 1; i <= n; ++i) ++cnt[rk[i] = s[i]];
  for (i = 1; i <= m; ++i) cnt[i] += cnt[i - 1];
  for (i = n; i >= 1; --i) sa[cnt[rk[i]]--] = i;

  for (w = 1; w < n; w <<= 1, m = p) {  // m=p 就是优化计数排序值域
    for (p = 0, i = n; i > n - w; --i) id[++p] = i;
    for (i = 1; i <= n; ++i)
      if (sa[i] > w) id[++p] = sa[i] - w;
    //memset(cnt, 0, sizeof(cnt));
    for(int i=0;i<=m;i++)cnt[i]=0;
    for (i = 1; i <= n; ++i) ++cnt[px[i] = rk[id[i]]];
    for (i = 1; i <= m; ++i) cnt[i] += cnt[i - 1];
    for (i = n; i >= 1; --i) sa[cnt[px[i]]--] = id[i];
    //memcpy(oldrk, rk, sizeof(rk));
    for(int i=0;i<=n;i++)oldrk[i]=rk[i];
    for (p = 0, i = 1; i <= n; ++i)
      rk[sa[i]] = cmp(sa[i], sa[i - 1], w) ? p : ++p;
  }

	
	  
}

cost(c,1)=cost(c∗1/c,1/c)∗c=cost(1,1/c)∗ccost(c, 1) = cost(c *1/c, 1/c) *c = cost(1, 1/c) * c cost(c,1)=cost(c∗1/c,1/c)∗c=cost(1,1/c)∗c

cost(u/v,1)=cost(1,v/u)∗u/vcost(u/v, 1) = cost(1, v/u) * u/v cost(u/v,1)=cost(1,v/u)∗u/v

第一位是每条边容量，第二位是总容量。
注意每条增广路的费用是依此递增的。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=2e6+20,M=8e6,inf=1e9;
int h[N],to[M],val[M],cost[M],nxt[M],dis[N],fr[N],pre[N],vis[N],cnt=1;
int s,t,x[N],y[N],w[N];
int k,n,m,C;
int ans[N],sum[N],si;
void  add(int x,int y,int v,int c)
{
    to[++cnt]=y;val[cnt]=v;cost[cnt]=c;nxt[cnt]=h[x];h[x]=cnt;
    to[++cnt]=x;val[cnt]=0;cost[cnt]=-c;nxt[cnt]=h[y];h[y]=cnt;
}
queue<int>q;
bool spfa()
{
    int x,i;
    memset(dis,0x3f,sizeof dis);
    memset(vis,0,sizeof vis);
    memset(fr,-1,sizeof fr);
    dis[s]=0;q.push(s);
    while(!q.empty())
    {
        x=q.front();q.pop();vis[x]=0;
        for(i=h[x];i;i=nxt[i])
        {
            if(val[i]&&dis[to[i]]>dis[x]+cost[i])
            {
            	dis[to[i]]=dis[x]+cost[i],fr[to[i]]=x,pre[to[i]]=i;
                if(!vis[to[i]])
                {
                q.push(to[i]);vis[to[i]]=1;
                }
            }
            
        }
    }
    return ~fr[t];
}
int mincost()
{
    int i,k;
    while(spfa())
    {
        k=inf;
        for(i=t;i!=s;i=fr[i])k=min(k,val[pre[i]]);
        ans[++si]=dis[t];sum[si]=sum[si-1]+ans[si];
        for(i=t;i!=s;i=fr[i])val[pre[i]]-=k,val[pre[i]^1]+=k;
    }
    return  1;
}
int main()
{
	while(~scanf("%d%d",&n,&m))
	{
		memset(ans,0,sizeof ans);
		memset(sum,0,sizeof sum);
		memset(h,0,sizeof h);
		cnt=1;C=1;si=0;
		for(int i=1;i<=m;i++)
		{
			scanf("%d%d%d",&x[i],&y[i],&w[i]);
			add(x[i],y[i],1,w[i]);
		} 
		s=1,t=n;
		mincost();
		int q;scanf("%d",&q);
		while(q--)
		{
			int u,v;scanf("%d%d",&u,&v);
			if(1ll*u*si<v){puts("NaN");continue;}
			ll w=v/u,ans1=1ll*sum[w]*u+1ll*ans[w+1]*(v%u),g=__gcd(ans1,1ll*v);
			//printf("%lld\n",ans1);
			printf("%lld/%lld\n",ans1/g,v/g);
		}
	}
}

#include<bits/stdc++.h>
using namespace std;
const int N=1010,M=3e5+200;
struct E{int to,nxt;}e[M];
int n,m,id[N],Id,ans,match[N],pre[N],vis[N];
int h[N],cnt,f[N],d[N];
int x[N],y[N],num[N][N];
queue<int>q;
void init(){
     Id=ans=cnt=0;
	memset(h,-1,sizeof h);
	memset(id,0,sizeof id);
	memset(match,0,sizeof match);
}
void add(int u,int v){ e[cnt]=E{v,h[u]},h[u]=cnt++; }
int getf(int x){ return f[x]==x?x:f[x]=getf(f[x]); }
int LCA(int x,int y)
{
	for(++Id;;swap(x,y))
		if(x)
		{
			x=getf(x);
			if(id[x]==Id) return x;  
			else id[x]=Id,x=pre[match[x]];		
		}
}
void blossom(int x,int y,int l)
{ 
	while(getf(x)!=l)
	{
		pre[x]=y,y=match[x];
		if(vis[y]==2) vis[y]=1,q.push(y);
		if(getf(x)==x) f[x]=l;
		if(getf(y)==y) f[y]=l;
		x=pre[y];
	} 
}
bool aug(int s){
	for(int i=1;i<=n;i++)
		vis[i]=pre[i]=0,f[i]=i;
	while(!q.empty()) q.pop();
	q.push(s),vis[s]=1;
	while(!q.empty())
	{
		int u=q.front();q.pop();
		for(int i=h[u];~i;i=e[i].nxt)
		{
			int v=e[i].to;
			if(getf(u)==getf(v)||vis[v]==2) continue;
			if(!vis[v])
			{
				vis[v]=2,pre[v]=u;
				if(!match[v])
				{ 
					for(int x=v,last;x;x=last)
						last=match[pre[x]],match[x]=pre[x],match[pre[x]]=x;		
					return 1;
				}	
				vis[match[v]]=1,q.push(match[v]);
			} 
			else 
			{
				 
				int lca=LCA(u,v);
				blossom(u,v,lca),blossom(v,u,lca);
			}
		}	
	}
	return 0;
}
void build()
{
	int kk=0;
	for(int i=1;i<=n;i++)
	    for(int j=1;j<=d[i];j++)
	    num[i][j]=++kk;
	for(int i=1;i<=m;i++){
		add(kk+1,kk+2);
		add(kk+2,kk+1);
		for(int j=1;j<=d[x[i]];j++){
			add(num[x[i]][j],kk+1);
			add(kk+1,num[x[i]][j]);
		}
		for(int j=1;j<=d[y[i]];j++){
			add(num[y[i]][j],kk+2);
			add(kk+2,num[y[i]][j]);
		}
		kk+=2;
	}
	n=kk;
}
int main()
{
	while(~scanf("%d%d",&n,&m))
	{
		init();
		for(int i=1;i<=n;i++)scanf("%d",&d[i]); 
		for(int i=1;i<=m;i++){
			scanf("%d%d",&x[i],&y[i]);
		}
		build();
		
		for(int i=1;i<=n;i++){
			if(!match[i]&&aug(i)){
				ans++;
			}
		}
		if(ans*2==n)puts("Yes");
		else puts("No");
	}
	return 0;
}

}
