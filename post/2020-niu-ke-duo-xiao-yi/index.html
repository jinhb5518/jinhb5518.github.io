拆点拆边带花树，一般图最大匹配（网络流是错的

#include<bits/stdc++.h>
using namespace std;
const int N=1010,M=3e5+200;
struct E{int to,nxt;}e[M];
int n,m,id[N],Id,ans,match[N],pre[N],vis[N];
int h[N],cnt,f[N],d[N];
int x[N],y[N],num[N][N];
queue<int>q;
void init(){
     Id=ans=cnt=0;
	memset(h,-1,sizeof h);
	memset(id,0,sizeof id);
	memset(match,0,sizeof match);
}
void add(int u,int v){ e[cnt]=E{v,h[u]},h[u]=cnt++; }
int getf(int x){ return f[x]==x?x:f[x]=getf(f[x]); }
int LCA(int x,int y)
{
	for(++Id;;swap(x,y))
		if(x)
		{
			x=getf(x);
			if(id[x]==Id) return x;  
			else id[x]=Id,x=pre[match[x]];		
		}
}
void blossom(int x,int y,int l)
{ 
	while(getf(x)!=l)
	{
		pre[x]=y,y=match[x];
		if(vis[y]==2) vis[y]=1,q.push(y);
		if(getf(x)==x) f[x]=l;
		if(getf(y)==y) f[y]=l;
		x=pre[y];
	} 
}
bool aug(int s){
	for(int i=1;i<=n;i++)
		vis[i]=pre[i]=0,f[i]=i;
	while(!q.empty()) q.pop();
	q.push(s),vis[s]=1;
	while(!q.empty())
	{
		int u=q.front();q.pop();
		for(int i=h[u];~i;i=e[i].nxt)
		{
			int v=e[i].to;
			if(getf(u)==getf(v)||vis[v]==2) continue;
			if(!vis[v])
			{
				vis[v]=2,pre[v]=u;
				if(!match[v])
				{ 
					for(int x=v,last;x;x=last)
						last=match[pre[x]],match[x]=pre[x],match[pre[x]]=x;		
					return 1;
				}	
				vis[match[v]]=1,q.push(match[v]);
			} 
			else 
			{
				 
				int lca=LCA(u,v);
				blossom(u,v,lca),blossom(v,u,lca);
			}
		}	
	}
	return 0;
}
void build()
{
	int kk=0;
	for(int i=1;i<=n;i++)
	    for(int j=1;j<=d[i];j++)
	    num[i][j]=++kk;
	for(int i=1;i<=m;i++){
		add(kk+1,kk+2);
		add(kk+2,kk+1);
		for(int j=1;j<=d[x[i]];j++){
			add(num[x[i]][j],kk+1);
			add(kk+1,num[x[i]][j]);
		}
		for(int j=1;j<=d[y[i]];j++){
			add(num[y[i]][j],kk+2);
			add(kk+2,num[y[i]][j]);
		}
		kk+=2;
	}
	n=kk;
}
int main()
{
	while(~scanf("%d%d",&n,&m))
	{
		init();
		for(int i=1;i<=n;i++)scanf("%d",&d[i]); 
		for(int i=1;i<=m;i++){
			scanf("%d%d",&x[i],&y[i]);
		}
		build();
		
		for(int i=1;i<=n;i++){
			if(!match[i]&&aug(i)){
				ans++;
			}
		}
		if(ans*2==n)puts("Yes");
		else puts("No");
	}
	return 0;
}

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;;
const int N=3e6+3000;
const ll mod=998244353;
ll f[N];
ll qpow(ll a,ll b){
	ll ans=1;
	while(b)
	{
		if(b&1)ans=(ans*a)%mod;
		b>>=1;
		a=(a*a)%mod;
	}
	return ans;
}
int main()
{
	ll n;f[0]=1;
	for(int i=1;i<=3e6+10;i++)f[i]=f[i-1]*i%mod;
	while(cin>>n)
	{
		cout<<f[n]*f[n]%mod*qpow(f[2*n+1],mod-2)%mod<<endl;
	}
	
}
