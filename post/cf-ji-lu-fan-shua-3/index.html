题意：给一个树，要求对树的每个边赋值，要求边权乘积等于k，值为一的边权在树中最小，要让树中任意两点的距离总和最大，给出k的质因数分解。
题解：首先知道每个边的权重为(n−sz[i])∗sz[i](n-sz[i])*sz[i](n−sz[i])∗sz[i]，其次注意不要取模后再排序。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int mod=1e9+7;
const int N=1e5+200;
int n,m;
vector<int>G[N];
ll sz[N],p[N],b[N];
void dfs(int u,int fa)
{
	sz[u]=1;
	for(auto v:G[u])
	{
		if(v!=fa){dfs(v,u);sz[u]+=sz[v];}
	}
}
int main()
{
  	int T;scanf("%d",&T);
  	while(T--)
  	{
  		
  	    ll ans=0;
		  scanf("%d",&n);
  	    for(int i=1;i<=n;i++)G[i].clear(),p[i]=1;
  	    for(int u,v,i=1;i<n;i++){
  	    	scanf("%d%d",&u,&v);
  	    	G[u].push_back(v);
  	    	G[v].push_back(u);
		}
		dfs(1,0);
		scanf("%d",&m);
		for(int i=1;i<=m;i++){
			scanf("%lld",&p[i]);
		}
		m=max(n-1,m);
		sort(p+1,p+1+m);
		if(m>n-1){
			for(int i=n;i<=m;i++)p[n-1]=(p[i]*p[n-1])%mod;
		}
		//sort(p+1,p+n);
		for(int i=2;i<=n;i++)b[i-1]=(n-sz[i])*sz[i];
		sort(b+1,b+n);
		for(int i=1;i<n;i++){
			ans=(ans+p[i]*b[i]%mod)%mod;
			//printf("%lld   %lld\n",p[i],b[i]);
		}
		printf("%lld\n",ans);
	}
}

题意：给定一个1e6*1e6的矩形，给若干条在与矩形相交一点垂直或水平的线段，问把矩形分成几个部分？
题解：首先，如果横跨左右肯定答案++，其次由于相交一点水平与垂直的线段若相交则必然答案++，树状数组维护即可。

#include<bits/stdc++.h>
#define fi first 
#define se second 
#define pb push_back 
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
const int mod=1e9+7;
const int M=1e6+1,N=1e6+10;
int n,m;
vector<pii> h[N],v[N];
int c[N];
void add(int x,int val){for(int i=x;i<N;i+=i&-i)c[i]+=val;} 
int ask(int x){int ans=0;for(int i=x;i>0;i-=i&-i)ans+=c[i];return ans;}
int main()
{
  	
	  	
		scanf("%d%d",&n,&m);
	  	ll ans=0;
	  	for(int i=1;i<=n;i++){
	  		int a,b,c;scanf("%d%d%d",&a,&b,&c);a++,b++,c++;
	  	    h[a].pb({b,c});
	  	    if(b==1&&c==M)ans++;   
		}
		for(int i=1;i<=m;i++){
	  		int x,d,u;scanf("%d%d%d",&x,&d,&u);x++,d++,u++;
	  	    v[d].pb({x,1});v[u+1].pb({x,-1});
	  	    if(d==1&&u==M)ans++;
	  	
		}
		for(int i=1;i<=M;i++)
		{
			for(auto e:v[i]){
				add(e.fi,e.se);
			}
			for(auto e:h[i]){  
				ans+=ask(e.se)-ask(e.fi-1);
			}
		}
		printf("%lld\n",ans+1);
	
	  
}
