题意：计算满足下述序列的数量
1.每个数组包含nnn个元素；
2.每个元素都是111到mmm之间的整数；
3.对于每个数组，只有一对相等的元素；
3.对于每个数组aaa，都有一个索引i，使得数组在第i个元素之前严格升序，在其第i个元素之后严格降序（正式而言，这意味着aj<aj+1a_j<a_{j+1}aj​<aj+1​，如果j<ij <ij<i，而aj>aj+1a_j> a_{j + 1}aj​>aj+1​，如果j≥ij\ge ij≥i）。
分析：因为一个数组里有n−1n-1n−1个元素,故需要从mmm个元素中选出n−1n-1n−1个，其次枚举那个重复的元素，因为最大的元素必不能是重复的元素，故有n−2n-2n−2种可能，最后考虑其余的元素应该放在最大元素的两边，故有2n−32^{n-3}2n−3种可能，所以最终答案为(n−2)Cmn−12n−3\left(n-2\right)C_m^{n-1}2^{n-3}(n−2)Cmn−1​2n−3
代码：

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll mod=998244353;
ll qpow(ll a,ll b){
	ll ans=1;
	while(b){
		if(b&1)ans=(ans*a)%mod;a=(a*a)%mod;b>>=1;
	}return ans;
}
ll n,m,r=1,ans=1;
int main()
{
	
	scanf("%lld%lld",&n,&m);
	if(n==2){
		puts("0");return 0;
	}
	for(ll i=n;i<=m;i++)ans=(ans*i)%mod;
	for(ll i=1;i<=m-n+1;i++)r=(r*i)%mod;
	ans=(ans*qpow(r,mod-2))%mod;
	ans=(ans*(n-2))%mod;
	ans=(ans*qpow(2,n-3))%mod;
	printf("%lld\n",ans);
	
}

题意：给定一个序列，如果有相邻的数相等ai=ai+1a_i=a_{i+1}ai​=ai+1​，则可以用ai+1a_i+1ai​+1取代，问最后能得到最小的序列长度是多少。
分析：区间dp，其中dp[l][r]表示区间l到r能被表示成一个数，值为dp[l][r],最后O(n2)O\left(n^2\right)O(n2)扫过即可

#include<bits/stdc++.h>
using namespace std;
const int N=1e3+20;
int n,a[N];
int f[N][N],g[N];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int len=1;len<=n;len++)
	{
		for(int l=1;l+len-1<=n;l++)
		{
			int r=l+len-1;
			if(l==r){
				f[l][l]=a[l];continue;
			}
			f[l][r]=-1;
			for(int k=l;k<r;k++)
			{
				if(f[l][k]==f[k+1][r]&&f[l][k]!=-1)
				{
					f[l][r]=f[l][k]+1;
				}
			}
		}
	}
	memset(g,0x3f,sizeof g);
	g[0]=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=i;j<=n;j++)
		{
			if(f[i][j]!=-1){
				g[j]=min(g[j],g[i-1]+1);
			}
		}
	}
	cout<<g[n]<<endl;
}

题意：给定一个长度为nnn的数组aia_iai​,你可以将任意aia_iai​变为ai−1a_i-1ai​−1，问在模h的意义下，前缀和数组sk=∑i=1kaks_k=\sum_{i=1}^k a_ksk​=∑i=1k​ak​落在[l,r][l,r][l,r]的数量最多？
分析：考虑dp，f[i][j]f[i][j]f[i][j]表示考虑到前iii个数，当前前缀和为jjj的落在[l,r][l,r][l,r]区间的个数，可以滚动数组优化一下。

#include<bits/stdc++.h>
using namespace std;
const int N=2e3+200;
typedef long long ll;
int a[N],n,h,l,r;
int f[N],g[N];
int main()
{
    cin>>n>>h>>l>>r;
    for(int i=1;i<=n;i++){
    	scanf("%d",&a[i]);
	}
	for(int i=0;i<h;i++)g[i]=-2147483647;
	g[0]=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=0;j<h;j++)
		{
			f[j]=max(g[(j-a[i]+h)%h],g[(j-a[i]+1+h)%h]);
			if(l<=j&&j<=r)f[j]++;
		}
		for(int j=0;j<h;j++)g[j]=f[j];
	}
	int ans=0;
	for(int i=0;i<h;i++)ans=max(ans,f[i]);
	cout<<ans<<endl;
	
}

题意：给定一棵树，结点用黑白染色，问以从111到nnn为根，白点与黑点的差最大的联通块的值。
分析：树形dp 统计每个点子树的贡献是简单的，本题的关键点在于如何统计每个点父亲方向的贡献，可以用父亲方向的子树贡献减去自身贡献加父亲的父亲方向贡献求解。

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=4e5+300,M=N;
typedef long long ll;
int h[N],nxt[M],to[M],val[M],cnt=0;
int n,a[N];
int  dp[N],ans[N];
void add(int u,int v)
{
    to[++cnt]=v,nxt[cnt]=h[u];h[u]=cnt;
}
void dfs(int u,int fa)
{
    dp[u]=a[u];
	for(int i=h[u];~i;i=nxt[i])
	{
		int v=to[i];if(v==fa)continue;
		dfs(v,u);
		dp[u]+=max(dp[v],0);
	}
}
void dfs_d(int u,int fa)
{
    //cout<<ans[u]<<endl;
	ans[u]=dp[u];
	for(int i=h[u];~i;i=nxt[i])
    {
		int v=to[i];
		if(v==fa)continue;
		dp[u]-=max(dp[v],0);
		dp[v]+=max(dp[u],0);
		dfs_d(v,u);
		dp[v]-=max(dp[u],0);
		dp[u]+=max(dp[v],0);
    }
}
int main()
{
    scanf("%d",&n);
    memset(h,-1,sizeof h);
    for(int i=1;i<=n;i++)
    {
      	scanf("%d",&a[i]);
		if(a[i]==0)a[i]=-1;
    } 
    for(int i=1;i<=n-1;i++)
       {
       	   int u,v;
		   scanf("%d%d",&u,&v);
       	   add(u,v),add(v,u);
	   }
    dfs(1,-1);
    dfs_d(1,-1);
    for(int i=1;i<=n;i++)
	{
	    printf("%d ",ans[i]);
	}
    
}
