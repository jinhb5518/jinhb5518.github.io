题意：计算满足下述序列的数量
1.每个数组包含nnn个元素；
2.每个元素都是111到mmm之间的整数；
3.对于每个数组，只有一对相等的元素；
3.对于每个数组aaa，都有一个索引i，使得数组在第i个元素之前严格升序，在其第i个元素之后严格降序（正式而言，这意味着aj<aj+1a_j<a_{j+1}aj​<aj+1​，如果j<ij <ij<i，而aj>aj+1a_j> a_{j + 1}aj​>aj+1​，如果j≥ij\ge ij≥i）。
分析：因为一个数组里有n−1n-1n−1个元素,故需要从mmm个元素中选出n−1n-1n−1个，其次枚举那个重复的元素，因为最大的元素必不能是重复的元素，故有n−2n-2n−2种可能，最后考虑其余的元素应该放在最大元素的两边，故有2n−32^{n-3}2n−3种可能，所以最终答案为(n−2)Cmn−12n−3\left(n-2\right)C_m^{n-1}2^{n-3}(n−2)Cmn−1​2n−3
代码：

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll mod=998244353;
ll qpow(ll a,ll b){
	ll ans=1;
	while(b){
		if(b&1)ans=(ans*a)%mod;a=(a*a)%mod;b>>=1;
	}return ans;
}
ll n,m,r=1,ans=1;
int main()
{
	
	scanf("%lld%lld",&n,&m);
	if(n==2){
		puts("0");return 0;
	}
	for(ll i=n;i<=m;i++)ans=(ans*i)%mod;
	for(ll i=1;i<=m-n+1;i++)r=(r*i)%mod;
	ans=(ans*qpow(r,mod-2))%mod;
	ans=(ans*(n-2))%mod;
	ans=(ans*qpow(2,n-3))%mod;
	printf("%lld\n",ans);
	
}

题意：给定一个序列，如果有相邻的数相等ai=ai+1a_i=a_{i+1}ai​=ai+1​，则可以用ai+1a_i+1ai​+1取代，问最后能得到最小的序列长度是多少。
分析：区间dp，其中dp[l][r]表示区间l到r能被表示成一个数，值为dp[l][r],最后O(n2)O\left(n^2\right)O(n2)扫过即可

#include<bits/stdc++.h>
using namespace std;
const int N=1e3+20;
int n,a[N];
int f[N][N],g[N];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int len=1;len<=n;len++)
	{
		for(int l=1;l+len-1<=n;l++)
		{
			int r=l+len-1;
			if(l==r){
				f[l][l]=a[l];continue;
			}
			f[l][r]=-1;
			for(int k=l;k<r;k++)
			{
				if(f[l][k]==f[k+1][r]&&f[l][k]!=-1)
				{
					f[l][r]=f[l][k]+1;
				}
			}
		}
	}
	memset(g,0x3f,sizeof g);
	g[0]=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=i;j<=n;j++)
		{
			if(f[i][j]!=-1){
				g[j]=min(g[j],g[i-1]+1);
			}
		}
	}
	cout<<g[n]<<endl;
}

题意：给定一棵树，结点用黑白染色，问以从111到nnn为根，白点与黑点的差最大的联通块的值。
分析：树形dp 统计每个点子树的贡献是简单的，本题的关键点在于如何统计每个点父亲方向的贡献，可以用父亲方向的子树贡献减去自身贡献加父亲的父亲方向贡献求解。

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=4e5+300,M=N;
typedef long long ll;
int h[N],nxt[M],to[M],val[M],cnt=0;
int n,a[N];
int  dp[N],ans[N];
void add(int u,int v)
{
    to[++cnt]=v,nxt[cnt]=h[u];h[u]=cnt;
}
void dfs(int u,int fa)
{
    dp[u]=a[u];
	for(int i=h[u];~i;i=nxt[i])
	{
		int v=to[i];if(v==fa)continue;
		dfs(v,u);
		dp[u]+=max(dp[v],0);
	}
}
void dfs_d(int u,int fa)
{
    //cout<<ans[u]<<endl;
	ans[u]=dp[u];
	for(int i=h[u];~i;i=nxt[i])
    {
		int v=to[i];
		if(v==fa)continue;
		dp[u]-=max(dp[v],0);
		dp[v]+=max(dp[u],0);
		dfs_d(v,u);
		dp[v]-=max(dp[u],0);
		dp[u]+=max(dp[v],0);
    }
}
int main()
{
    scanf("%d",&n);
    memset(h,-1,sizeof h);
    for(int i=1;i<=n;i++)
    {
      	scanf("%d",&a[i]);
		if(a[i]==0)a[i]=-1;
    } 
    for(int i=1;i<=n-1;i++)
       {
       	   int u,v;
		   scanf("%d%d",&u,&v);
       	   add(u,v),add(v,u);
	   }
    dfs(1,-1);
    dfs_d(1,-1);

题意：给定一个数列，其中保证每一个数因子数≤7\le 7≤7,求长度最小子序列乘积为完全平方数，输出最小长度，没有则输出−1-1−1。
分析：由题可知每个数的质因子个数必≤3\le 3≤3,并且质因子的偶次方是没有用的，可以直接去掉，这样就剩下三种可能1,p,pq1,p,pq1,p,pq如果出现111直接输出1即可，否则建图,ppp的话与111建边，pqpqpq的话ppp与qqq建边,bfs找最小环便是答案。注意本题中我们只需要对所有权值 ≤maxai\le \sqrt{max a_i}≤maxai​​的节点搜索，因为权值 >maxai>\sqrt{max a_i}>maxai​​的节点只可能出现 1 次，必定为奇数不能成环。

#include<bits/stdc++.h>
using namespace std;
const int N=1e6+200;
int a[N],ans=2e9,n;
vector<int>e[N],fac;
int dis[N];
bool vis[N];
void init(int &x)
{
	int t=x;
	for(int i=2;i*i<=t;i++)
	{
		if(x%i==0)
		{
			int cnt=0;
			while(x%i==0)x/=i,cnt++;
			if(cnt&1){
				x*=i;
			}
		}
	}
}
void bfs(int x)
{
	for(auto u:fac)dis[u]=2e9,vis[u]=0;
	dis[x]=0;vis[x]=true;
	queue<int>q;q.push(x);
	while(!q.empty()){
		int top=q.front();q.pop();
	    vis[top]=false;
		for(auto i:e[top]){
			if(dis[i]>dis[top]+1){
				dis[i]=dis[top]+1;
				vis[i]=true;
				q.push(i);
			}
			else if(vis[i])ans=min(ans,dis[i]+dis[top]+1);
		}
	}
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
    	scanf("%d",&a[i]);
    	init(a[i]);
	}
	sort(a+1,a+1+n);
	if(a[1]==1){
		puts("1");return 0;
	}
	for(int i=1;i<n;i++)
	{
		if(a[i]==a[i+1]){
			puts("2");return 0;
		}
	}
	for(int i=1;i<=n;i++){
		int tmp[2]={-1,-1};
		tmp[0]=a[i];
		for(int j=2;j*j<=a[i];j++)
		{
			if(a[i]%j)continue;
			tmp[0]=j;
			if(a[i]/j!=1)tmp[1]=a[i]/j;
			break;
		}
		if(tmp[1]==-1)tmp[1]=1;
		e[tmp[0]].push_back(tmp[1]);
		e[tmp[1]].push_back(tmp[0]);
		fac.push_back(tmp[1]);
		fac.push_back(tmp[0]);
	}
	sort(fac.begin(),fac.end());
	fac.erase(unique(fac.begin(),fac.end()),fac.end());
	for(int i=1;i<=1000;i++)
	{
		if(e[i].size()==0)continue;
		bfs(i);
		//printf("%d  %d\n",i,ans);
	}
	if(ans==2e9)puts("-1");
	else printf("%d\n",ans);
}

题意：给定一个nnn个点mmm条边的无向图。要求找出一个点数恰好为 ⌈n⌉\left\lceil\sqrt{n}\right\rceil⌈n​⌉ 的独立集，或者找出一个点数 ≥⌈n⌉\ge\left\lceil\sqrt{n}\right\rceil≥⌈n​⌉ 的环。
分析：环：通过dfs树找返祖边，如果点数≥⌈n⌉\ge\left\lceil\sqrt{n}\right\rceil≥⌈n​⌉直接输出。
否则的话，再考虑独立集的构造，我们用 ⌈n⌉−1\left\lceil\sqrt{n}\right\rceil-1⌈n​⌉−1 种颜色，给 dfs 树上的节点按照 dep[i]%(⌈n⌉−1)dep[i]\%(\left\lceil\sqrt{n}\right\rceil-1)dep[i]%(⌈n​⌉−1) 的规则染色，由于 dfs 树的特性，相同深度的节点必定构成一个独立集，而在模意义下相同的任意两个不同深度节点之间，如果存在直连边就说明存在一个节点数 ≥⌈n⌉\ge\left\lceil\sqrt{n}\right\rceil≥⌈n​⌉ 的环。因此我们只需要找到一个点数 ≥⌈n⌉\ge\left\lceil\sqrt{n}\right\rceil≥⌈n​⌉ 的集合并恰好输出 ⌈n⌉\left\lceil\sqrt{n}\right\rceil⌈n​⌉ 个点。

#include<bits/stdc++.h>
using namespace std;
const int N=2e5+200;
vector<int> e[N];
int n,m,lim,num[N],fa[N],dep[N];
bool ok=false;
void dfs(int u){
	++num[dep[u]%(lim-1)];
	for(auto v:e[u])
	{
		if(v==fa[u])continue;
		if(dep[v]==-1)
		{
			dep[v]=dep[u]+1;
			fa[v]=u;
			dfs(v);
		}
		else if(dep[u]-dep[v]>=lim-1)
		{
			ok=true;
			puts("2");
			printf("%d\n",dep[u]-dep[v]+1);
			for(int i=u;i!=fa[v];i=fa[i])printf("%d ",i);
			exit(0);
		}
	}
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0;i<=n;i++)dep[i]=-1;
    dep[1]=0;
    for(int i=1;i<=m;i++)
    {
    	int x,y;scanf("%d%d",&x,&y);
    	e[x].push_back(y);
    	e[y].push_back(x);
	}
	lim=ceil(sqrt(1.0*n));
	dfs(1);
	if(ok==false)
	{
		puts("1"); 
		for(int i=0;i<lim-1;i++)
		{
		    if(num[i]<lim)continue;
			int tot=lim;
			for(int j=1;j<=n;j++){
				if(dep[j]%(lim-1)==i)
				{
					printf("%d ",j);
					tot--;
					if(!tot)return 0;
				}
			}	
		}
	}
}

题意：给定一个长度为nnn的数组,ai≤1e7a_i\le 1e7ai​≤1e7，求(a1+a2)⊕(a1+a3)⊕...(an−1+an)( a_1+a_2)\oplus( a_1+a_3)\oplus...(a_{n-1}+a_n)(a1​+a2​)⊕(a1​+a3​)⊕...(an−1​+an​)
分析：首先因为ai≤1e7a_i\le 1e7ai​≤1e7，故所求≤2e7\le 2e7≤2e7,然后按位考虑贡献，由于每个数都在式子里出现n−1n-1n−1次，故只有当n−1n-1n−1与当前位是111的个数同时为奇数时才对答案产生贡献，之后考虑进位，先把每个数取模这一位都取出来，排序，用双指针法找出当前两个数相加进位有多少对 ，若为奇数对答案产生贡献。

#include<bits/stdc++.h>
using namespace std;
const int N=4e5+200;
int n,a[N],sum,b[N],ans;
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    for(int B=1;B<2e7;B<<=1)
    {
    	for(int j=1;j<=n;j++)b[j]=(a[j]&(B-1));
    	int sum=0;
    	for(int i=1;i<=n;i++)sum+=((a[i]&B)!=0);
    	if((sum&1)&&((n-1)&1))ans^=B;
    	sort(b+1,b+1+n);
    	int L=1,R=n;sum=0;
    	for(L=1;L<=n;L++)
    	{
    		R=max(L,R);
    		for(;R>L&&b[L]+b[R]>=B;R--);
    		sum^=n-R;
		}
		if(sum&1)ans^=B;
	}
	printf("%d\n",ans);
}
