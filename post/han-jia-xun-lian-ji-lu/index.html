题意：给定一个长度为nnn的KKK进制数在KKK进制表示下是回文串且能被ppp整除的数有多少个，其中ppp为质数。其中 n≤1e18n\le 1e18n≤1e18 p≤1000p\le 1000p≤1000 K≤16K\le 16K≤16。
题解：首先考虑 ppp 与 kkk 不互素的情况。这种情况下，整除只与最后一位有关，且有 kp−1\frac{k}{p}−1pk​−1 种取法，其余位任取。注意 k=pk = pk=p 时答案为 [n=1][n = 1][n=1]，可能会导致部分没有特判的程序挂掉。 现在考虑(p,k)=1(p,k) = 1(p,k)=1 的情况。考虑暴力：dpi,jdp_{i,j}dpi,j​ 表示前 i 位和后 i 位填好了，其余位取 0 时 mod p=jp = jp=j 的方案数。更新一位的复杂度是 O(kp)O(kp)O(kp) 的，总复杂度是 O(nkp)O(nkp)O(nkp) 的。 考虑优化，由费马小定理，kik^iki mod ppp 存在长度为 p−1p−1p−1 循环节，因此 ki+kn−i−1k^i + k^{n−i−1}ki+kn−i−1mod ppp 也有长度为 p−1p−1p−1 的循环节。按 p−1p−1p−1 分块后，每块的贡献是相同的，而块的背包合并可以通过类似多项式乘法的循环卷积来实现，因此就可以快速幂了。 预处理一个块和处理边缘的零散部分的复杂度是 O(kp2)O(kp^2)O(kp2), 一次块的暴力卷积是 O(p2)O(p^2)O(p2) 的，因此总复 杂度为 O(kp2+p2(logn−logp))O(kp^2 + p^2(logn−logp))O(kp2+p2(logn−logp))。 标程使用 vector 实现，不开 O2O2O2 也只跑了一秒，应该是不卡常的。(来自官方题解)
代码：

#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
typedef pair<int, int> pii;
#define read(a) scanf("%d", &a)
#define x first
#define y second

#define MOD 998244353

LL power(LL x, LL y, LL z) {
	LL ret = 1;
	for ( x %= z; y; y >>= 1) {
		if (y & 1) ret = ret * x % z;
		x = x * x % z;
	}
	return ret;
}

LL n;
int p, K;

void init(vector <int> & a) {
	a.clear();
	a.resize(p);
	a[0] = 1;
}

void upd(vector <int> &a, LL b, int st = 0) {
	int x;//一单元量 
	if (b == n - b - 1) {
		x = int(power(K, b, p)) % p;
	} else {
		x = int(power(K, b, p) + power(K, n - b - 1, p)) % p;
	}
	vector <int> c;
	c.resize(p);
	for (int i = st; i < K; i++) {
		for (int j = 0; j < p; j++) {
			(c[(j + i * x) % p] += a[j]) %= MOD;
		}
	}
	swap(a, c);
}

void mul(vector <int> &a, vector <int> &b) {
	vector <int> c;
	c.resize(p);
	for (int i = 0; i < p; i++)
		for (int j = 0; j < p; j++) {
			int x = (i + j) % p;
			c[x] = int((c[x] + (LL)a[i] * b[j]) % MOD);
		}
	swap(a, c);
}

void power(vector <int> &a, long long b) {
	vector <int> ret;
	init (ret);
	for ( ; b; b >>= 1 ) {
		if (b & 1) mul(ret, a);
		mul(a, a);
	}
	swap(a, ret);
}

vector <int> ans, tr;

int main() {
	scanf("%lld%d%d", &n, &p, &K);
	if (n == 1) {
		int aa = 0;
		for (int i = 0; i < K; i++) {
			if (i % p == 0) aa ++;
		}
		cout << aa << endl;
		return 0;
	}
	LL sz = n / 2 - 1;//除去个位和最高位配对 
	if (K % p == 0) {
		cout << (K / p - 1) * power(K, sz + (n & 1), MOD) % MOD << endl;//减一是因为不能为K 
		return 0;
	}
	init(ans);
	init(tr);
	LL tot = sz / (p - 1); //p-1分成一块 
	if (tot) {
		for (int i = 2; i <= p; i++) {//枚举前p-1位 
			upd(tr, i - 1);
		}
		power(tr, tot);
		mul(ans, tr);
	}
	upd(ans, 0, 1);
	for (long long i = 2 + tot * (p - 1); i <= (n + 1) / 2; i++) {
		upd(ans, i - 1);//最后几位 
	}
	printf("%d\n", ans[0]);
}
