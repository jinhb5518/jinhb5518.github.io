A ABB

题意：给定一个字符串，求最少在结尾添加多少个字符串使之成为回文串。
分析：用马拉车求出包含结尾的最长回文子串用原串长度减去即可。

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=31000010;
int n,ans,hw[N];
char a[N],s[N<<1];
void init()
{
	s[0]=s[1]='#';
	for(int i=0;i<n;i++)
	{
		s[i*2+2]=a[i];
		s[i*2+3]='#';
	}
	n=n*2+2;
	s[n]=0;
}
void manacher()
{
      int mx=0,mid;
	  for(int i=1;i<n;i++)
	  {
	  	if(i<mx)hw[i]=min(hw[(mid<<1)-i],hw[mid]+mid-i);
	  	else hw[i]=1;
	  	for(;s[i+hw[i]]==s[i-hw[i]];++hw[i]);
	  	
	  	if(hw[i]+i>mx)
	  	{
	  		mx=hw[i]+i;
	  		mid=i;
		}
      }
}

int main()
{
	int l;
	scanf("%d",&n);
	scanf("%s",a);l=n;
	n=strlen(a);
	init();
	manacher();
	ans=1;
	for(int i=0;i<n;i++)if(i+hw[i]==n)ans=max(hw[i]-1,ans);
	printf("%d",l-ans);
	return 0;
}

C Bob in Wonderland

题意：给定一棵树，问最多经过多少次操作将其变成一条链。
分析：对于会非根节点，儿子个数大于1会增加贡献，根节点儿子个数大于2会增加贡献。
(或直接算每个点入度大于2则增加贡献

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=1e6+200;
vector<int>g[N];
int son[N];
long long ans;
void dfs(int u,int fa){
    for(auto v:g[u]){
        if(v==fa)continue;
        dfs(v,u);son[u]++;
    }
    if(u==1)ans+=max(0,son[u]-2);
    else ans+=max(0,son[u]-1);
}
int main()
{
    int n;scanf("%d",&n);
    for(int i=1;i<n;i++)
    {
        int a,b;scanf("%d%d",&a,&b);g[a].push_back(b);
        g[b].push_back(a);
    }
    for(int i=1;i<=n;i++)if(g[i].size()>2)
    ans+=g[i].size()-2;
    printf("%lld",ans);
}

D Deep800080

题意：给定一条直线，n个点，问圆心在直线上半径为R且能覆盖最多个点的个数。
分析：点能被圆心覆盖当且仅当以该点为圆心做半径为R的圆与直线相交的线段区间内，故只要将所有线段全求出来起始坐标加一末尾坐标减一找最大即可。

#include<bits/stdc++.h>
using namespace std;
int n, r, A, B, ans, cnt, res;
struct node
{
    int x, y;
}p[1<<20];
pair<double, int> q[1<<20];
inline bool cmp(pair<double, int> a, pair<double, int> b)
{
    return a.first == b.first ? a.second > b.second : a.first < b.first;
}
int main()
{
    scanf("%d%d%d%d", &n, &r, &A, &B);
    for(int i = 0; i < n; i++)
        scanf("%d%d", &p[i].x, &p[i].y);
    if(A == 0)
    {
        swap(A, B);
        for(int i = 0; i < n; i++)
            swap(p[i].x, p[i].y);
    }
    for(int i = 0; i < n; i++)
    {
        double x = A * (1.0 * A * p[i].x + 1.0 * B * p[i].y) / (1.0 * A * A + 1.0 * B * B);
        double y = x * B / A;
        double d = sqrt((p[i].x - x) * (p[i].x - x) + (p[i].y - y) * (p[i].y - y));
        if(d <= r)
        {
            double p = sqrt(1.0*r*r - d*d);
            double len = sqrt(x*x+y*y);
            if(x<0)len = -len;
            q[cnt++] = make_pair(len+p, -1);
            q[cnt++] = make_pair(len-p, 1);
        }
    }
    sort(q, q+cnt, cmp);
    for(int i = 0; i < cnt; i++)
        res += q[i].second, ans = max(ans, res);
    printf("%d\n", ans);
}

E Zeldain Garden

题意：输入n,m , 计算[n,m]范围内的每个数的因子个数的总和。n,m<=1e12
分析：通过观察可得，[1,n]中每个数的因子个数的总和等于∑i=1nn/i\sum\limits_{i=1}^nn/ii=1∑n​n/i
整除分块或者转化为根号求一下即可

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=1e6+200;
typedef long long ll;
ll solve(ll k)
{
	ll ans=0,t=sqrt(k);
	for(int i=1;i<=t;i++)ans+=k/i;
	ans=ans*2-t*t;
	return ans;
}
int main()
{
	ll n,m;
	scanf("%lld%lld",&n,&m);
    printf("%lld",solve(m)-solve(n-1));	
}

F Light Emitting Hindenburg

题意：给定n个数，求最少k个数的按位与（2进制）最大
分析：从高位到低位枚举如果该位的1的个数超过k个，则将剩余的n-k个排除，依此递推下去。

#include<bits/stdc++.h>
using namespace std;
const int N=2e5+200;
int a[N];
bool vis[N];
int main()
{
    int n,k;scanf("%d%d",&n,&k);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    for(int i=29;i>=0;i--)
    {
    	int cnt=0;
		for(int j=1;j<=n;j++)if(!vis[j]&&((a[j]>>i)&1))cnt++;  
		if(cnt>=k){
			for(int j=1;j<=n;j++)if(!vis[j]&&(!((a[j]>>i)&1)))vis[j]=1;
		}       
	}
	int res=-1;
	for(int i=1;i<=n;i++){
		if(!vis[i])res&=a[i];
	}
	cout<<res<<endl;
}

G K==S

题意：给定q个字符串，问长度为n且不包含其中任意一个字符串的可能性种数
题解：原题poj2778，AC自动机加矩阵快速幂，去掉所有终止节点建邻接矩阵，n次方即为答案，细节简代码。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=2010,mod=1e9+7;
int tot,n,qq;
int c[N][26],d[N],q[2*N],ta,he,f[N];
struct mat{
	ll a[101][101];
	void clear(){
		memset(a,0,sizeof a);
	}
}U;
mat mul(mat a,mat b)
{
	mat c;
	c.clear();
	for(int i=0;i<=tot;i++){
		for(int j=0;j<=tot;j++)
		{
		    for(int k=0;k<=tot;k++)
			{
			   c.a[i][j]=(c.a[i][j]+a.a[i][k]*b.a[k][j]%mod)%mod;	
			}	
		}
	}
	return c;
}
int main(){
	cin>>n>>qq;
	while(qq--){
		int l;string s;
		cin>>l>>s;
		int i,j;
		for(i=j=0;i<l;j=c[j][s[i++]-'a'])if(!c[j][s[i]-'a'])c[j][s[i]-'a']=++tot;
		d[j]=1;
	}
	for(int i=0;i<26;i++)if(c[0][i])q[ta++]=c[0][i];
	while(he!=ta){
		int i=q[he++];
		for(int j=0;j<26;j++)if(c[i][j])d[c[i][j]]|=d[f[q[ta++]=c[i][j]]=c[f[i]][j]];
		else c[i][j]=c[f[i]][j];
	}
	U.clear();
	for(int i=0;i<=tot;i++)
	{
		if(d[i])continue;
		for(int j=0;j<26;j++)
		{
			if(d[c[i][j]])continue;
			U.a[i][c[i][j]]++;
		}
	}
	mat ans;
	ans.clear();
	for(int i=0;i<=tot;i++)ans.a[i][i]=1;
	while(n){
		//printf("%d\n",n);
		if(n&1)ans=mul(ans,U);
		n>>=1;
		U=mul(U,U);
		
	}
	ll res=0;
	for(int i=0;i<=tot;i++)res=(res+ans.a[0][i])%mod;
	cout<<res<<endl;
	return 0;
	
}

H Ponk Warshall

题意：给两个只有四个字母组成的字符串，问第一个需要交换多少次才能到第二个。
分析：仔细考虑一下只有下几种情况1.交换一次改变两个
2.交换两次改变三个
3.交换三次改变四个
操作编号越小越优

#include<bits/stdc++.h>
using namespace std;
const int N=1e6+200;
int f[500][500];
char str[4]={'A','C','G','T'};
int main()
{
    string s,t;
    cin>>s>>t;
    int n=s.length();
    for(int i=0;i<n;i++){
    	if(s[i]!=t[i])f[s[i]][t[i]]++;
	}
	int ans=0;
	for(int i=0;i<4;i++)
	   for(int j=0;j<4;j++)
	   {
	   	    int t=min(f[str[i]][str[j]],f[str[j]][str[i]]);
	   	    ans+=t;
	   	    f[str[i]][str[j]]-=t;
	   	    f[str[j]][str[i]]-=t;
	   }
	for(int i=0;i<4;i++)
	   for(int j=0;j<4;j++)
	   for(int z=0;z<4;z++)
	   {
	   	    int t=min({f[str[i]][str[j]],f[str[j]][str[z]],f[str[z]][str[i]]});
	   	    ans+=2*t;
	   	    f[str[i]][str[j]]-=t;
	   	    f[str[j]][str[z]]-=t;
	   	    f[str[z]][str[i]]-=t;
	   }
	for(int i=0;i<4;i++)
	   for(int j=0;j<4;j++)
	   for(int z=0;z<4;z++)
	   for(int k=0;k<4;k++)
	   {
	   	    int t=min({f[str[i]][str[j]],f[str[j]][str[z]],f[str[z]][str[k]],f[str[k]][str[i]]});
	   	    ans+=3*t;
	   	    f[str[i]][str[j]]-=t;
	   	    f[str[j]][str[z]]-=t;
	   	    f[str[z]][str[k]]-=t;
	   	    f[str[k]][str[i]]-=t;
	   }
	   printf("%d",ans);
	
}

I Saba1000kg

题意：给定一个n个点e条边的无向图，q个询问每次给出m个点，问每次给出的点构成了多少个连通块。n,e,q<=1e5 ∑m≤1e5\sum m \le 1e5∑m≤1e5
分析：求连通块用并查集，记录点的度数，小于n\sqrt{n}n​的直接暴力处理，将大于n\sqrt{n}n​的将所有能到的点保存在一个集合里，查询时直接找这些点是否在同一个集合里即可。(这东西叫根号分治？

#include<bits/stdc++.h>
using namespace std;
const int N=1e5+200,K=400;
vector<int>g[N];
int n,e,p,v[N],d[N],fa[N],vv[N];
bool vis[N];
set<int>st[N];
int find(int x){
	return x==fa[x]?fa[x]:fa[x]=find(fa[x]);
}
int main()
{
	scanf("%d%d%d",&n,&e,&p);
	for(int i=1;i<=e;i++){
		int u,v;
		scanf("%d%d",&u,&v);
		g[u].push_back(v);g[v].push_back(u);
		d[u]++;d[v]++;
	}
	for(int i=1;i<=n;i++){
		fa[i]=i;
		if(d[i]>K){
			for(auto v:g[i])st[i].insert(v);
		}
	}
	while(p--)
	{
		
		int m;scanf("%d",&m);
		for(int i=1;i<=m;i++)
		{
			scanf("%d",&v[i]);
			vis[v[i]]=1;
		}
		int cnt=0;
		for(int i=1;i<=m;i++){
			if(d[v[i]]<=K)
			{
				for(auto w:g[v[i]]){
					if(vis[w]){
						int a=find(w);
						int b=find(v[i]);
						if(a!=b){
							fa[a]=b;
						}
					}
				}
			}
			else
			{
				vv[++cnt]=v[i];
			}
		}
		for(int i=1;i<=cnt;i++){
			int x=vv[i];
			for(int j=i+1;j<=cnt;j++)if(st[x].count(vv[j])){
				int y=vv[j];
				int a=find(x);
				int b=find(y);
				if(a!=b){
					fa[a]=b;
				}
			}
		}
		int ans=0;
		for(int i=1;i<=m;i++)
		{
			if(v[i]==fa[v[i]])ans++;
		}
		printf("%d\n",ans);
		for(int i=1;i<=m;i++){
			vis[v[i]]=0;
			fa[v[i]]=v[i];
		}
	}
}
