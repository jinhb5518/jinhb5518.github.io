A

题意：给定一个字符串，求最少在结尾添加多少个字符串使之成为回文串。
分析：用马拉车求出包含结尾的最长回文子串用原串长度减去即可。

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=31000010;
int n,ans,hw[N];
char a[N],s[N<<1];
void init()
{
	s[0]=s[1]='#';
	for(int i=0;i<n;i++)
	{
		s[i*2+2]=a[i];
		s[i*2+3]='#';
	}
	n=n*2+2;
	s[n]=0;
}
void manacher()
{
      int mx=0,mid;
	  for(int i=1;i<n;i++)
	  {
	  	if(i<mx)hw[i]=min(hw[(mid<<1)-i],hw[mid]+mid-i);
	  	else hw[i]=1;
	  	for(;s[i+hw[i]]==s[i-hw[i]];++hw[i]);
	  	
	  	if(hw[i]+i>mx)
	  	{
	  		mx=hw[i]+i;
	  		mid=i;
		}
      }
}

int main()
{
	int l;
	scanf("%d",&n);
	scanf("%s",a);l=n;
	n=strlen(a);
	init();
	manacher();
	ans=1;
	for(int i=0;i<n;i++)if(i+hw[i]==n)ans=max(hw[i]-1,ans);
	printf("%d",l-ans);
	return 0;
}

题意：给定一棵树，问最多经过多少次操作将其变成一条链。
分析：对于会非根节点，儿子个数大于1会增加贡献，根节点儿子个数大于2会增加贡献。
(或直接算每个点入度大于2则增加贡献

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=1e6+200;
vector<int>g[N];
int son[N];
long long ans;
void dfs(int u,int fa){
    for(auto v:g[u]){
        if(v==fa)continue;
        dfs(v,u);son[u]++;
    }
    if(u==1)ans+=max(0,son[u]-2);
    else ans+=max(0,son[u]-1);
}
int main()
{
    int n;scanf("%d",&n);
    for(int i=1;i<n;i++)
    {
        int a,b;scanf("%d%d",&a,&b);g[a].push_back(b);
        g[b].push_back(a);
    }
    for(int i=1;i<=n;i++)if(g[i].size()>2)
    ans+=g[i].size()-2;
    printf("%lld",ans);
}

题意：输入n,m , 计算[n,m]范围内的每个数的因子个数的总和。n,m<=1e12
分析：通过观察可得，[1,n]中每个数的因子个数的总和等于∑i=1nn/i\sum\limits_{i=1}^nn/ii=1∑n​n/i
整除分块或者转化为根号求一下即可

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=1e6+200;
typedef long long ll;
ll solve(ll k)
{
	ll ans=0,t=sqrt(k);
	for(int i=1;i<=t;i++)ans+=k/i;
	ans=ans*2-t*t;
	return ans;
}
int main()
{
	ll n,m;
	scanf("%lld%lld",&n,&m);
    printf("%lld",solve(m)-solve(n-1));	
}

F

题意：给定n个数，求最少k个数的按位与（2进制）最大
分析：从高位到低位枚举如果该位的1的个数超过k个，则将剩余的n-k个排除，依此递推下去。

#include<bits/stdc++.h>
using namespace std;
const int N=2e5+200;
int a[N];
bool vis[N];
int main()
{
    int n,k;scanf("%d%d",&n,&k);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    for(int i=29;i>=0;i--)
    {
    	int cnt=0;
		for(int j=1;j<=n;j++)if(!vis[j]&&((a[j]>>i)&1))cnt++;  
		if(cnt>=k){
			for(int j=1;j<=n;j++)if(!vis[j]&&(!((a[j]>>i)&1)))vis[j]=1;
		}       
	}
	int res=-1;
	for(int i=1;i<=n;i++){
		if(!vis[i])res&=a[i];
	}
	cout<<res<<endl;
}

H

题意：给两个只有四个字母组成的字符串，问第一个需要交换多少次才能到第二个。
分析：仔细考虑一下只有下几种情况1.交换一次改变两个
2.交换两次改变三个
3.交换三次改变四个
操作编号越小越优

#include<bits/stdc++.h>
using namespace std;
const int N=1e6+200;
int f[500][500];
char str[4]={'A','C','G','T'};
int main()
{
    string s,t;
    cin>>s>>t;
    int n=s.length();
    for(int i=0;i<n;i++){
    	if(s[i]!=t[i])f[s[i]][t[i]]++;
	}
	int ans=0;
	for(int i=0;i<4;i++)
	   for(int j=0;j<4;j++)
	   {
	   	    int t=min(f[str[i]][str[j]],f[str[j]][str[i]]);
	   	    ans+=t;
	   	    f[str[i]][str[j]]-=t;
	   	    f[str[j]][str[i]]-=t;
	   }
	for(int i=0;i<4;i++)
	   for(int j=0;j<4;j++)
	   for(int z=0;z<4;z++)
	   {
	   	    int t=min({f[str[i]][str[j]],f[str[j]][str[z]],f[str[z]][str[i]]});
	   	    ans+=2*t;
	   	    f[str[i]][str[j]]-=t;
	   	    f[str[j]][str[z]]-=t;
	   	    f[str[z]][str[i]]-=t;
	   }
	for(int i=0;i<4;i++)
	   for(int j=0;j<4;j++)
	   for(int z=0;z<4;z++)
	   for(int k=0;k<4;k++)
	   {
	   	    int t=min({f[str[i]][str[j]],f[str[j]][str[z]],f[str[z]][str[k]],f[str[k]][str[i]]});
	   	    ans+=3*t;
	   	    f[str[i]][str[j]]-=t;
	   	    f[str[j]][str[z]]-=t;
	   	    f[str[z]][str[k]]-=t;
	   	    f[str[k]][str[i]]-=t;
	   }
	   printf("%d",ans);
	
}
