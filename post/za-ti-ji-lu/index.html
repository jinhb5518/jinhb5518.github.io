#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=2e5+200,mod=1e9+7;
vector<int> g[N];
int n;

ll sz[N],tmp=1,a[N];
ll qpow(ll a,ll b)
{
	ll ans=1;
	while(b){
		if(b&1)ans=(ans*a)%mod;
		a=a*a%mod;
		b>>=1;
	}
	return ans;
}
void dfs(int u,int fa)
{
	sz[u]=1;
	for(auto to:g[u])
	{
		if(to==fa)continue;
		dfs(to,u);
		sz[u]+=sz[to];
	}
	tmp=(tmp*sz[u])%mod;
}
void dfs2(int u,int fa)
{
	for(auto to:g[u])
	{
		if(to==fa)continue;
		a[to]=a[u]*qpow(sz[to],mod-2)%mod*(n-sz[to])%mod;
		dfs2(to,u);
	}
}
int main()
{
    scanf("%d",&n);
    ll lc=1;
    for(ll i=2;i<=n;i++)
    {
    	int a,b;
		scanf("%d%d",&a,&b);
    	g[a].push_back(b);
    	g[b].push_back(a);
    	lc=(lc*i)%mod;
	}
	dfs(1,-1);
	a[1]=tmp;
	dfs2(1,-1);
	for(int i=1;i<=n;i++)printf("%lld\n",lc*qpow(a[i],mod-2)%mod);
}

AC自动机

#include<bits/stdc++.h>
using namespace std;
int n,nn,l,c[2010][26],q[2005],g[1006][2006],m,d[2005],ta,he,f[2005],ans;
char s[2010];
int main()
{
	scanf("%d%d",&nn,&l);
	for(int k,z=0;z<nn;z++)
	{
		scanf("%s%d",s,&k);
		m=strlen(s);
		int j,i;
		for(i=j=0;i<m;j=c[j][s[i++]-'a'])if(!c[j][s[i]-'a'])c[j][s[i]-'a']=++n;
		d[j]+=k;
	}
	for(int i=0;i<26;i++)if(c[0][i])q[ta++]=c[0][i];
	while(he!=ta){
		int i=q[he++];
		for(int j=0;j<26;j++)if(c[i][j])d[c[i][j]]+=d[f[q[ta++]=c[i][j]]=c[f[i]][j]];
		else c[i][j]=c[f[i]][j];
	}
	memset(g,128,sizeof g);
	ans=g[0][0];
	//printf("%d \n",ans);
	g[0][0]=0;
	for(int i=0;i<l;i++)for(int j=0;j<=n;j++)for(int k=0;k<26;k++)g[i+1][c[j][k]]=max(g[i+1][c[j][k]],g[i][j]+d[c[j][k]]);
	for(int i=0;i<=n;i++)ans=max(ans,g[l][i]);
	printf("%d\n",ans);
	return 0;
}
